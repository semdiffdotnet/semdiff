90  private async static Task<IEnumerable<Diagnostic>> OnCompilationAsync(Compilation comp)
91  {
92  Logger.Trace($"Entering {nameof(OnCompilationAsync)}: {comp.AssemblyName}");
93  try
94  {
95  var data = comp.GetRepoAndTrees(GetRepo);
96  foreach (var repo in data.Keys)
97  {
98  try
99  {
100  await repo.UpdateRemoteChangesAsync();
101  }
102  catch (GitHubAuthenticationFailureException ex)
103  {
104  return new[] { Diagnostics.AuthenticationFailure(ex.GithubMessage) };
105  }
106  catch (GitHubRateLimitExceededException ex)
107  {
108  return new[] { Diagnostics.RateLimit(ex.Limit, ex.Authenticated) };
109  }
110  catch (GitHubUrlNotFoundException ex)
111  {
112  return new[] { Diagnostics.NotGitHubRepo(ex.Path) };
113  }
114  catch (GitHubUnknownErrorException ex)
115  {
116  return new[] { Diagnostics.UnexpectedError($"Communicating with GitHub", ex.Message) };
117  }
118  catch (GitHubDeserializationException ex)
119  {
120  return new[] { Diagnostics.UnexpectedError($"Deserializing Data", ex.Message) };
121  }
122  catch (Exception ex)
123  {
124  Logger.Error($"Unhandled Exception from {nameof(repo.UpdateRemoteChangesAsync)}: {ex.GetType().Name}: {ex.Message} << {ex.StackTrace} >>");
125  return new[] { Diagnostics.UnexpectedError($"Communicating with GitHub", ex.Message) };
126  }
127  }
128 
129  var diagnostics = new List<Diagnostic>();
130  foreach (var rt in data)
131  {
132  var repo = rt.Key;
133  foreach (var t in rt.Value) //Foreach tree
134  {
135  diagnostics.AddRange(Analyze(comp.GetSemanticModel(t), repo));
136  }
137  }
138  return diagnostics;
139  }
140  catch (Exception ex)
141  {
142  Logger.Error($"Unhandled Exception: {ex.GetType().Name}: {ex.Message} << {ex.StackTrace} >>");
143  return new[] { Diagnostics.UnexpectedError("Running Analysis", ex.Message) };
144  }
145  finally
146  {
147  Logger.Trace($"Entering {nameof(OnCompilationAsync)}: {comp.AssemblyName}");
148  }
149  }

151  private static IEnumerable<Diagnostic> Analyze(SemanticModel semanticModel, Repo repo)
152  {
153  Logger.Trace($"Entering {nameof(Analyze)}: {semanticModel?.SyntaxTree?.FilePath}");
154 
155  var fps = Analysis.ForFalsePositive(repo, semanticModel.SyntaxTree);
156  var fns = Analysis.ForFalseNegative(repo, semanticModel);
157 #if DEBUG
158  fps = fps.Log(fp => Logger.Info($"{fp}"));
159  fns = fns.Log(fn => Logger.Info($"{fn}"));
160 #endif
161  var diags = fps.Select(Diagnostics.Convert).Concat(fns.Select(Diagnostics.Convert));
162 
163  Logger.Trace($"Exiting {nameof(Analyze)}: {semanticModel?.SyntaxTree?.FilePath}");
164  return diags;
165  }
