136  public void GetCurrentSaved()
137  {
138  Debug.Assert(this != null);
139  try
140  {
141  if (!Directory.Exists(CacheDirectory))
142  return;
143  if (!File.Exists(CachedLocalPullRequestListPath))
144  return;
145  var json = File.ReadAllText(CachedLocalPullRequestListPath);
146  var list = JsonConvert.DeserializeObject<IEnumerable<PullRequest>>(json);
147  PullRequests.Clear();
148  foreach (var p in list)
149  {
150  //Restore Self-Referential Loops
151  p.ParentRepo = this;
152  foreach (var r in p.ValidFiles)
153  {
154  r.ParentPullRequst = p;
155  }
156 
157  if (VerifyPullRequestCache(p))
158  {
159  PullRequests.Add(p);
160  foreach (var r in p.ValidFiles)
161  {
162  r.LoadFromCache();
163  }
164  }
165  else
166  {

168  Directory.Delete(p.CacheDirectory, true);

174  catch (Exception ex)

201  public async Task<IList<PullRequest>> GetPullRequestsAsync()
202  {
203  var url = $"/repos/{Owner}/{RepoName}/pulls";
204  var etag = Ref.Create(EtagNoChanges);
205  var updated = await GetPaginatedListAsync<PullRequest>(url, etag);
206  var outdated = PullRequests;
207  EtagNoChanges = etag.Value;
208  if (updated == null)
209  {
210  return null;
211  }
212  if (outdated != null)
213  {
214  var prToRemove = outdated.Where(old => updated.All(newpr => newpr.Number != old.Number));
215  foreach (var pr in updated)
216  {
217  pr.ParentRepo = this;
218  var old = outdated.FirstOrDefault(o => o.Number == pr.Number);
219  if (old != null)
220  {
221  pr.LastWrite = old.LastWrite;
222  }
223  }
224  DeletePRsFromDisk(prToRemove);
225  }
226  await Task.WhenAll(updated.Select(async pr =>
227  {
228  var files = await GetPaginatedListAsync<RepoFile>($"/repos/{Owner}/{RepoName}/pulls/{pr.Number}/files");
229  foreach (var f in files)
230  {
231  f.ParentPullRequst = pr;
232  }
233  pr.Files = PullRequest.FilterFiles(files).ToList();
234  return pr;
235  }));
236  PullRequests.Clear();
237  PullRequests.AddRange(updated);
238  UpdateLocalSavedList();
239  return updated;
240  }

432  internal async Task<string> HttpGetAsync(string url, Ref<string> etag = null, Ref<string> pages = null)
433  {
434  //Request, but retry once waiting 5 minutes
435  Client.DefaultRequestHeaders.IfNoneMatch.Clear();
436  if (etag?.Value != null)
437  {
438  Client.DefaultRequestHeaders.IfNoneMatch.Add(EntityTagHeaderValue.Parse(etag.Value));
439  }
440  var response = await Client.GetAsync(url);
441  IEnumerable<string> headerVal;
442  if (response.Headers.TryGetValues("X-RateLimit-Limit", out headerVal))
443  {
444  RequestsLimit = int.Parse(headerVal.Single());
445  }
446  if (response.Headers.TryGetValues("X-RateLimit-Remaining", out headerVal))
447  {
448  RequestsRemaining = int.Parse(headerVal.Single());
449  }
450  if (!response.IsSuccessStatusCode)
451  {
452  switch (response.StatusCode)
453  {
454  case HttpStatusCode.Unauthorized:
455  var unauth = await response.Content.ReadAsStringAsync();
456  var unauthorizedError = DeserializeWithErrorHandling<GitHubError>(unauth, supress_error: true);
457  Logger.Error($"{nameof(GitHubAuthenticationFailureException)}: {unauthorizedError?.Message ?? unauth}");
458  throw new GitHubAuthenticationFailureException(unauthorizedError.Message);
459  case HttpStatusCode.Forbidden:
460  var forbid = await response.Content.ReadAsStringAsync();
461  var forbidError = DeserializeWithErrorHandling<GitHubError>(forbid, supress_error: true);
462  Logger.Error($"{nameof(GitHubRateLimitExceededException)}: {forbidError?.Message ?? forbid}");
463  throw new GitHubRateLimitExceededException(RequestsLimit, !string.IsNullOrWhiteSpace(AuthToken));
464  case HttpStatusCode.NotModified:
465  //Returns null because we have nothing to update if nothing was modified
466  return null;
467 
468  default:
469  var str = await response.Content.ReadAsStringAsync();
470  if (str == "Not Found")
471  throw new GitHubUnknownErrorException("Not Found");
472  var error = DeserializeWithErrorHandling<GitHubError>(str, supress_error: true);
473  throw error?.ToException() ?? new GitHubUnknownErrorException(str);
474  }
475  }
476  if (etag != null && response.Headers.TryGetValues("ETag", out headerVal))
477  {
478  etag.Value = headerVal.Single();
479  }
480  if (pages != null)
481  {
482  pages.Value = response.Headers.TryGetValues("Link", out headerVal) ? ParseNextLink(headerVal) : null;
483  }
484  return await response.Content.ReadAsStringAsync();
485  }

487  private static Repo AddRepo(string repoDir)
488  {
489  using (var r = new Repository(repoDir))
490  {
491  var matchingUrls = r.Network.Remotes
492  .Select(remote => _gitHubUrl.Match(remote.Url))
493  .Where(m => m.Success);
494  var match = matchingUrls.FirstOrDefault();
495  if (match == null)
496  {
497  Logger.Error(nameof(GitHubUrlNotFoundException));
498  throw new GitHubUrlNotFoundException(path: repoDir);
499  }
500 
501  var url = match.Value.Trim();
502  var owner = match.Groups[3].Value.Trim();
503  var name = match.Groups[4].Value.Trim();
504  if (name.EndsWith(".git"))
505  {
506  name = name.Substring(0, name.Length - 4);
507  }
508  Logger.Debug($"Repo: Owner='{owner}' Name='{name}' Url='{url}'");
509  var repo = new Repo(repoDir, owner, name);
510  repo.GetCurrentSaved();
511  return repo;
512  }
