22  public static IEnumerable<DetectedFalsePositive> ForFalsePositive(Repo repo, SyntaxTree local)
23  {
24  var relativePath = GetRelativePath(repo.LocalRepoDirectory, local.FilePath);
25  var pulls = GetPulls(repo, relativePath);
26  foreach (var pull in pulls)
27  {
28  var conflicts = Diff3.Compare(pull.File.BaseTree, local, pull.File.HeadTree);
29  var locs = GetInsertedMethods(conflicts.Local);
30  var rems = GetInsertedMethods(conflicts.Remote);
31 
32  var methodConflicts = conflicts.Conflicts.Where(con => con.Ancestor.Node is MethodDeclarationSyntax);
33 
34  foreach (var c in methodConflicts)
35  {
36  var ancestor = (MethodDeclarationSyntax)c.Ancestor.Node;
37 
38  var conflict = GetInnerMethodConflicts(ancestor, c.Remote, c.Local, locs, InnerMethodConflict.Local.Moved);
39 
40  //If the local was not moved, the local could have still been changed
41  if (conflict == null)
42  {
43  conflict = GetInnerMethodConflicts(ancestor, c.Local, c.Remote, rems, InnerMethodConflict.Local.Changed);
44  if (conflict == null)
45  {
46  continue;
47  }
48  }
49 
50  if (!conflict.DiffResult.Conflicts.Any(con => TriviaCompare.IsSemanticChange(con.Local.Node, con.Remote.Node)))
51  {
52  yield return new DetectedFalsePositive
53  {
54  Location = Location.Create(local, conflict.GetLocal().Identifier.Span),
55  MethodName = ancestor.Identifier.ToString(),
56  RemoteFile = pull.File,
57  RemoteChange = pull.Change,
58  ConflictType = conflict.LocalLocation == InnerMethodConflict.Local.Changed
59  ? DetectedFalsePositive.ConflictTypes.LocalMethodChanged
60  : DetectedFalsePositive.ConflictTypes.LocalMethodRemoved,
61  };
62  }
63  }
64  }

109  //Given the base class and a file that contains another version of the file
110  private static IEnumerable<Diff> DiffClassVersion(ClassDeclarationSyntax b, RepoFile f)
111  {
112  var ancestorDecs = f.BaseTree
113  .GetRoot()
114  .DescendantNodes()
115  .OfType<ClassDeclarationSyntax>();
116 
117  var remoteDecs = f.HeadTree
118  .GetRoot()
119  .DescendantNodes()
120  .OfType<ClassDeclarationSyntax>();
121 
122  var merged = MergeClassDeclarationSyntaxes(ancestorDecs, remoteDecs)
123  .FirstOrDefault(t => AreSameClass(t.Item1, b));
124 
125  return Diff.Compare(merged.Item1, merged.Item2)
126  .Where(d => TriviaCompare.IsSemanticChange(d.Ancestor.Node, d.Changed.Node));
127  }

177  private static InnerMethodConflict GetInnerMethodConflicts(MethodDeclarationSyntax ancestor, SpanDetails changed, SpanDetails removed, IEnumerable<MethodDeclarationSyntax> insertedMethods, InnerMethodConflict.Local type)
178  {
179  if (!TriviaCompare.IsSpanInNodeTrivia(removed.Span, removed.Node))
180  {
181  return null;
182  }
183  var change = changed.Node as MethodDeclarationSyntax;
184  if (change == null)
185  {
186  return null;
187  }
188  var moved = GetMovedMethod(insertedMethods, change);
189  if (moved == null)
190  {
191  return null;
192  }
193  var diffRes = Diff3.Compare(ancestor, moved, change);
194  return new InnerMethodConflict(ancestor, change, moved, diffRes, type);
195  }

197  private static MethodDeclarationSyntax GetMovedMethod(IEnumerable<MethodDeclarationSyntax> insertedMethods, MethodDeclarationSyntax change)
198  {
199  //A unique method seems to be defined by its name and its parameter types (including type params)
200 
201  var namesMatch = insertedMethods.Where(method => method.Identifier.Text == change.Identifier.Text);
202  //Note: will cause problems if parts of the signature were changed, including renaming params
203  //Note: this is a textual checking, it may or may not be better to do only compare the order and type of the parameters (same for type params)
204  var paramsMatch = namesMatch.Where(method => AreSame(method.ParameterList, change.ParameterList));
205  var typeParamsMatch = paramsMatch.Where(m => AreSame(m.TypeParameterList, change.TypeParameterList));
206 
207  return typeParamsMatch.SingleOrDefault();
208  }

234  private static IEnumerable<Tuple<ClassDeclarationSyntax, ClassDeclarationSyntax>>
235  MergeClassDeclarationSyntaxes(IEnumerable<ClassDeclarationSyntax> left,
236  IEnumerable<ClassDeclarationSyntax> rightE)
237  {
238  var right = rightE.ToList();
239  foreach (var l in left)
240  {
241  //Name of class should give a reasonably good idea of if they are the same class
242  //This will ignore any renaming of the class
243  var r = right.FirstOrDefault(rc => AreSameClass(rc, l));
244  if (r != null)
245  {
246  right.Remove(r);
247  yield return Tuple.Create(l, r);
248  }
249  }
250  }
