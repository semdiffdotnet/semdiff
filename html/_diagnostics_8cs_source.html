205  public static Diagnostic Convert(DetectedFalsePositive fps)
206  {
207  if (fps.ConflictType == DetectedFalsePositive.ConflictTypes.LocalMethodRemoved)
208  {
209  return Diagnostic.Create(SD0001, fps.Location, fps.MethodName, fps.RemoteChange.Title, fps.RemoteChange.Number);
210  }
211  else if (fps.ConflictType == DetectedFalsePositive.ConflictTypes.LocalMethodChanged)
212  {
213  return Diagnostic.Create(SD0002, fps.Location, fps.MethodName, fps.RemoteChange.Title, fps.RemoteChange.Number);
214  }
215  throw new NotImplementedException(fps.ConflictType.ToString());
216  }
