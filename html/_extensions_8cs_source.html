37  public static Queue<T> GetMergedQueue<T>(this IEnumerable<T> left, IEnumerable<T> right, Func<T, int> selector)
38  {
39  if (selector == null)
40  throw new InvalidOperationException();
41  var leftQueue = left.ToQueue();
42  var rightQueue = right.ToQueue();
43  var result = new Queue<T>();
44  while (true)
45  {
46  var leftEmpty = leftQueue.Count == 0;
47  var rightEmpty = rightQueue.Count == 0;
48  if (leftEmpty && rightEmpty)
49  {
50  break;
51  }
52  else if (leftEmpty)
53  {
54  result.Enqueue(rightQueue.Dequeue());
55  }
56  else if (rightEmpty)
57  {
58  result.Enqueue(leftQueue.Dequeue());
59  }
60  else
61  {
62  var l = selector(leftQueue.Peek());
63  var r = selector(rightQueue.Peek());
64  if (l > r)
65  {
66  result.Enqueue(rightQueue.Dequeue());
67  }
68  else
69  {
70  result.Enqueue(leftQueue.Dequeue());
71  }
72  }
73  }
74  return result;
75  }
